from sklearn.metrics import f1_score, accuracy_score, roc_auc_score, classification_report

# Lower threshold to 0.3 for better sensitivity
threshold = 0.3

lr_pred_adj  = (lr_prob  >= threshold).astype(int)
rf_pred_adj  = (rf_prob  >= threshold).astype(int)
xgb_pred_adj = (xgb_prob >= threshold).astype(int)

print("=== ADJUSTED RESULTS (threshold=0.3) ===\n")

models = ['Logistic Regression', 'Random Forest', 'XGBoost']
preds  = [lr_pred_adj, rf_pred_adj, xgb_pred_adj]
probs  = [lr_prob, rf_prob, xgb_prob]

results = {'Model': [], 'Accuracy': [], 'AUC Score': [], 'F1 Score': []}

for name, pred, prob in zip(models, preds, probs):
    acc = accuracy_score(y_test, pred)
    auc = roc_auc_score(y_test, prob)
    f1  = f1_score(y_test, pred)
    results['Model'].append(name)
    results['Accuracy'].append(round(acc, 4))
    results['AUC Score'].append(round(auc, 4))
    results['F1 Score'].append(round(f1, 4))
    print(f"--- {name} ---")
    print(classification_report(y_test, pred))

import pandas as pd
results_df = pd.DataFrame(results)
print(results_df.to_string(index=False))
